    \item \lstinline{A ; B} \alert{ne signifie pas} \og exécuter \lstinline{A} puis \lstinline{B} \fg

    \end{itemize}

void increment (Register<Integer> X) {
  Transaction t = new STMTransaction();
  while (!t.isCommited()) {
    try {
      t.begin();
      X.write(t, X.read(t) + 1);
      t.try_to_commit();
    } catch (AbortException e) {}
  }
}
